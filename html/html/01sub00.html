이미지 스프라이트

이미지 스프라이트 기술의 구체적인 내용이 담긴 테이블
설명 각각의 이미지파일로 만드는 방식이 아니라 특정 간격을 기준으로 하나의 파일 안에 여러 이미지들을 담고 CSS background-position 속성으로 위치를 조절하여 보여주는 방식

footer구역 fixed 문제해결

footer구역 fixed 기술의 구체적인 내용이 담긴 테이블
설명 브라우저 창이 footer 의 최소 크기보다 작을 경우 내용이 가려지는 현상
해결 방법 브라우저 창이 footer 의 최소 크기보다 작아질 경우 position : static, 커질 경우 position : fixed 로 처리함

/* footer fixed-static, opacity 이벤트 */ 
var footerFxOpEVent = function () {

      $winHt = $(window).height(); 
      $footerHt = $footer.innerHeight() - 1;

            $footer.css("position", "static");

         } 
      }; 
      thisEvent();

parallax scrolling

페럴렉스 스크롤링 기술의 구체적인 내용이 담긴 테이블
설명 스크롤 속도와 위치를 기준으로 이벤트가 발생하는 효과
원리 스크롤이 + 값일 시 header 의 opacity 가 + 로, 스크롤이 - 값일 시 main 의 opacity 가 - 되는 이벤트
화면
코드 
                                            
/* 스크롤 opacity 이벤트 */ 
var mainOpcChangeEvent = function () { 
    winHtControlEvent(); 
    var $scrollValue = $(window).scrollTop(); 
    var $sss = $scrollValue / $winHt; 
    
    if ($sss <= 1) { 
        $("#tit_header").css("opacity", 1 - $sss); 
        $("#tit_main").css("opacity", $sss); 
    } else { 
        $("#tit_header").css("opacity", 0); 
        $("#tit_main").css("opacity", 1); 
    } 
}; 
mainOpcChangeEvent(); 

$(window).on("scroll", function () { 
    mainOpcChangeEvent(); 
    $(window).on("resize", function () { 
        clearTimeout(timer2); timer2 = setTimeout(function () { 
            mainOpcChangeEvent(); 
        }, sec); 
    }); 
});
